FUNCIÓN BUSCAR_ISAM(clave_buscada, indice_primario, archivo_primario, archivo_overflow)  
    // Paso 1: Buscar en el índice primario para encontrar el bloque correspondiente  
    bloque = BUSCAR_EN_INDICE(clave_buscada, indice_primario)  
      
    // Paso 2: Buscar secuencialmente dentro del bloque primario  
    registro = BUSCAR_EN_BLOQUE(clave_buscada, bloque, archivo_primario)  
      
    SI (registro != NULL) ENTONCES  
        RETORNAR registro  // Registro encontrado en área primaria  
    SINO  
        // Paso 3: Si no está en el área primaria, buscar en overflow  
        registro_overflow = BUSCAR_EN_OVERFLOW(clave_buscada, archivo_overflow)  
          
        SI (registro_overflow != NULL) ENTONCES  
            RETORNAR registro_overflow  // Registro encontrado en overflow  
        SINO  
            RETORNAR NULL  // Registro no existe  
        FIN SI  
    FIN SI  
FIN FUNCIÓN  

// --- Funciones auxiliares ---  

FUNCIÓN BUSCAR_EN_INDICE(clave, indice)  
    // Busca en el índice primario el bloque donde debería estar la clave  
    PARA cada entrada EN indice HACER  
        SI (clave >= entrada.clave_min Y clave <= entrada.clave_max) ENTONCES  
            RETORNAR entrada.bloque_asociado  
        FIN SI  
    FIN PARA  
    RETORNAR NULL  // Clave fuera de rango (no debería pasar si el índice está bien construido)  
FIN FUNCIÓN  

FUNCIÓN BUSCAR_EN_BLOQUE(clave, bloque, archivo)  
    // Busca secuencialmente dentro de un bloque de datos  
    PARA cada registro EN bloque DEL archivo HACER  
        SI (registro.clave == clave) ENTONCES  
            RETORNAR registro  
        FIN SI  
    FIN PARA  
    RETORNAR NULL  // No encontrado  
FIN FUNCIÓN  

FUNCIÓN BUSCAR_EN_OVERFLOW(clave, archivo_overflow)  
    // Busca en el área de overflow (puede ser una lista enlazada o estructura similar)  
    PARA cada registro EN archivo_overflow HACER  
        SI (registro.clave == clave) ENTONCES  
            RETORNAR registro  
        FIN SI  
    FIN PARA  
    RETORNAR NULL  // No encontrado en overflow  
FIN FUNCIÓN
