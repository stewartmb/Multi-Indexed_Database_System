create table destinos (
    id int primary key index hash,
    name varchar[25] index seq,
    latitud double,
    longitud double,
    ciudad varchar[20] index bptree(3,2),
    pais varchar[20]
);

create index on destinos using bptree(pais) params (2,3);

create index on destinos using rtree(pais);

create index on destinos using seq(pais) params (2);
